var orange = {
    weight:"200g",
    color:"orange",
    taste:"sour"
};
var car = {
    brand:"Jaguar",
    price:"$80000",
    model:"XFL"
}
function mainJs(a){
    //请在此处编写代码
    /*********begin*********/
    if (a in orange) {
        return Object.keys(orange).join('');

        return Object.keys(car).join('');
    }
    return '';
    /*********end*********/
}
